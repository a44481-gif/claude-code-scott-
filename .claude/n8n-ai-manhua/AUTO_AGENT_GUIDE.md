# 🔧 方案三：n8n全自动工作流

## n8n工作流节点

```json
{
  "nodes": [
    {
      "name": "定时触发",
      "type": "scheduleTrigger",
      "parameters": {
        "rule": {
          "interval": [
            { "hour": 8 }
          ]
        }
      }
    },
    {
      "name": "生成内容",
      "type": "code",
      "parameters": {
        "jsCode": "// 随机选择题材\nconst themes = ['龙王','逆袭','虐渣','重生','神医','系统'];\nconst theme = themes[Math.floor(Math.random()*themes.length)];\nreturn { theme, prompt: `生成${theme}题材漫剧内容JSON` };"
      }
    },
    {
      "name": "Claude API",
      "type": "httpRequest",
      "parameters": {
        "url": "https://api.anthropic.com/v1/messages",
        "method": "POST",
        "body": {
          "model": "claude-sonnet-4-6",
          "max_tokens": 2048,
          "messages": [{"role": "user", "content": "{{ $json.prompt }}"}]
        },
        "headers": {
          "x-api-key": "YOUR_API_KEY",
          "anthropic-version": "2023-06-01"
        }
      }
    },
    {
      "name": "解析JSON",
      "type": "code",
      "parameters": {
        "jsCode": "// 解析Claude响应\nconst content = $input.first().json.content[0].text;\nconst json = JSON.parse(content.match(/```json\\n([\\s\\S]*?)\\n```/)[1]);\nreturn json;"
      }
    },
    {
      "name": "Discord通知",
      "type": "discord",
      "parameters": {
        "channel": "YOUR_DISCORD_WEBHOOK",
        "text": "🎬 新内容已生成\n题材：{{ $json.theme }}\n标题：{{ $json.title }}"
      }
    }
  ]
}
```
