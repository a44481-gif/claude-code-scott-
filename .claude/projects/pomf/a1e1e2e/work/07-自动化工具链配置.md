## 四、核心自动化脚本

### 4.1 主程序入口

```python
# main.py
import schedule
import time
from datetime import datetime
from src.content.script_generator import ScriptGenerator
from src.video.digital_human import DigitalHumanCreator
from src.publish.platform_poster import PlatformPoster
from src.analytics.report_generator import ReportGenerator
from src.email.report_sender import ReportSender

class AIVideoAgent:
    def __init__(self):
        self.script_gen = ScriptGenerator()
        self.digital_human = DigitalHumanCreator()
        self.poster = PlatformPoster()
        self.report_gen = ReportGenerator()
        self.email_sender = ReportSender()

    def daily_task(self):
        """每日任务：生成脚本 → 制作视频 → 发布"""
        print(f"[{datetime.now()}] 执行每日任务...")

        # 1. 生成今日脚本
        script = self.script_gen.generate_daily_script()
        print(f"✅ 脚本生成完成: {script['title']}")

        # 2. 制作数字人视频
        video_path = self.digital_human.create_video(script)
        print(f"✅ 视频制作完成: {video_path}")

        # 3. 发布到平台
        results = self.poster.publish_video(video_path, script)
        print(f"✅ 多平台发布完成: {results}")

    def weekly_task(self):
        """每周任务：生成报告 → 发送邮件"""
        print(f"[{datetime.now()}] 执行每周任务...")

        # 1. 收集数据
        data = self.report_gen.collect_weekly_data()

        # 2. 生成报告
        report_html = self.report_gen.generate_html_report(data)

        # 3. 发送邮件
        self.email_sender.send_report(report_html)
        print(f"✅ 周报已发送至 h13751019800@163.com")

    def run(self):
        """启动定时任务"""
        # 每日 06:00 执行
        schedule.every().day.at("06:00").do(self.daily_task)

        # 每周一 09:00 执行
        schedule.every().monday.at("09:00").do(self.weekly_task)

        print("🎬 AI 短视频运营 Agent 已启动...")
        while True:
            schedule.run_pending()
            time.sleep(60)

if __name__ == "__main__":
    agent = AIVideoAgent()
    agent.run()
```

# src/email/report_sender.py
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from datetime import datetime
import json

class ReportSender:
    def __init__(self):
        with open("config/email_config.json", "r") as f:
            self.config = json.load(f)

    def send_report(self, html_content):
        """发送报告邮件"""
        # 邮件主题
        end_date = datetime.now().strftime("%Y-%m-%d")
        start_date = (datetime.now() - timedelta(days=7)).strftime("%Y-%m-%d")
        subject = f"{self.config['report']['subject_prefix']}{start_date} 至 {end_date}"

        # 创建邮件
        msg = MIMEMultipart("alternative")
        msg["Subject"] = subject
        msg["From"] = f"{self.config['sender']['name']} <{self.config['sender']['email']}>"
        msg["To"] = self.config["recipient"]["email"]

        # 添加 HTML 内容
        msg.attach(MIMEText(html_content, "html", "utf-8"))

        # 发送邮件
        try:
            with smtplib.SMTP(self.config["smtp"]["server"], self.config["smtp"]["port"]) as server:
                if self.config["smtp"].get("use_tls", True):
                    server.starttls()
                server.login(self.config["sender"]["email"], self.config["sender"]["password"])
                server.send_message(msg)

            print(f"✅ 邮件发送成功: {subject}")
            return True
        except Exception as e:
            print(f"❌ 邮件发送失败: {e}")
            return False

## 六、Docker 部署（可选）

```dockerfile
# Dockerfile
FROM python:3.10-slim

WORKDIR /app

COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

COPY . .

CMD ["python", "main.py"]
```

```yaml
# docker-compose.yml
version: '3.8'

services:
  ai-video-agent:
    build: .
    environment:
      - HEYGEN_API_KEY=${HEYGEN_API_KEY}
      - ELEVENLABS_API_KEY=${ELEVENLABS_API_KEY}
      - TIKTOK_ACCESS_TOKEN=${TIKTOK_ACCESS_TOKEN}
    volumes:
      - ./outputs:/app/outputs
      - ./config:/app/config
    restart: unless-stopped
    crontab:
      - "0 6 * * * python /app/main.py --daily"
      - "0 9 * * 1 python /app/main.py --weekly"
```

## 七、使用指南

### 7.1 快速开始

```bash
# 1. 克隆/创建项目
mkdir ai-video-agent && cd ai-video-agent

# 2. 安装依赖
pip install -r requirements.txt

# 3. 配置 API Key
export HEYGEN_API_KEY="your-heygen-key"
export ELEVENLABS_API_KEY="your-elevenlabs-key"
export TIKTOK_ACCESS_TOKEN="your-tiktok-token"

# 4. 运行主程序
python main.py
```

### 7.2 单独执行任务

```bash
# 单独执行每日任务
python main.py --daily

# 单独执行周报任务
python main.py --weekly

# 生成单个视频
python main.py --generate --script-id SP-20260405
```

### 7.3 查看日志

```bash
# 查看运行日志
tail -f outputs/logs/agent.log

# 查看错误日志
grep ERROR outputs/logs/agent.log
```
